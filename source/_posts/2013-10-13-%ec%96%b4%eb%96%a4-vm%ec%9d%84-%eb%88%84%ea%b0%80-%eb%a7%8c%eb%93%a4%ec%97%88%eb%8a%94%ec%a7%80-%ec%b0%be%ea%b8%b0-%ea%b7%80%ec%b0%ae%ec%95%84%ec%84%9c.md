    #!/bin/bash  
    ip=$1
    
    port_id=$(quantum port-list | grep $ip | awk '{print $2}')  
    device_id=$(quantum port-show $port_id | awk '/device_id/{print $4}')  
    user_id=$(nova show $device_id | awk '/user_id/{print $4}')  
    keystone user-get $user_id
